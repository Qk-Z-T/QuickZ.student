                // সাধারণ টাইপিং মিসটেক ঠিক করা
                text = String(text)

                const hasBengali = /[\u0980-\u09FF]/.test(text); // চেক করবে বাংলা আছে কিনা

                if (hasMathDelimiters) {

            loadMoreRankings: async (examId, totalQuestions, reset = false) => {
                const myRouteId = window.currentRouteId;
                const c = document.getElementById('app-container');
                const uid = auth.currentUser.uid;

                let rankQuery;
                if (reset || !window.leaderboardLastVisible) {
                    rankQuery = query(
                        collection(db, "attempts"), 
                        where("examId", "==", examId),
                        orderBy("score", "desc"),
                        orderBy("submittedAt", "asc"),
                        limit(20)
                    );
                } else {
                    rankQuery = query(
                        collection(db, "attempts"), 
                        where("examId", "==", examId),
                        orderBy("score", "desc"),
                        orderBy("submittedAt", "asc"),
                        startAfter(window.leaderboardLastVisible),
                        limit(20)
                    );
                }

                try {
                    const snap = await getDocs(rankQuery);
                    if (myRouteId !== window.currentRouteId) return;

                    if (snap.empty) {
                        window.leaderboardHasMore = false;
                    } else {
                        window.leaderboardLastVisible = snap.docs[snap.docs.length - 1];
                    }

                    const newAttempts = [];
                    snap.forEach(d => newAttempts.push({ id: d.id, ...d.data() }));

                    // Compute time taken and sort (though already sorted by query)
                    newAttempts.forEach(a => {
                        if (a.startedAt && a.submittedAt) {
                            try {
                                const diffMs = a.submittedAt.toDate() - a.startedAt.toDate();
                                a.timeTakenSeconds = Math.floor(diffMs / 1000);
                            } catch (e) {
                                a.timeTakenSeconds = Infinity;
                            }
                        } else {
                            a.timeTakenSeconds = Infinity;
                        }
                    });

                    // Merge with existing if not reset
                    let allAttempts = reset ? newAttempts : [...(window.currentRankAttempts || []), ...newAttempts];

                    // Sort again to be safe
                    allAttempts.sort((a, b) => {
                        if (a.score !== b.score) return (b.score || 0) - (a.score || 0);
                        if (a.timeTakenSeconds !== b.timeTakenSeconds) return (a.timeTakenSeconds || Infinity) - (b.timeTakenSeconds || Infinity);
                        const aTime = a.submittedAt ? a.submittedAt.toDate().getTime() : 0;
                        const bTime = b.submittedAt ? b.submittedAt.toDate().getTime() : 0;
                        return aTime - bTime;
                    });

                    window.currentRankAttempts = allAttempts;

                    let rows = '';
                    let myRank = 'N/A';
                    let highestScore = 0;

                    allAttempts.forEach((a, i) => {
                        const isMe = a.userId === uid;
                        if(isMe) myRank = i + 1;
                        const currentScore = parseFloat(a.score) || 0;
                        if(currentScore > highestScore) highestScore = currentScore;

                        let timeDisplay = 'N/A';
                        if (a.timeTakenSeconds && a.timeTakenSeconds !== Infinity) {
                            const mins = Math.floor(a.timeTakenSeconds / 60);
                            const secs = a.timeTakenSeconds % 60;
                            timeDisplay = `${mins}m ${secs}s`;
                        }

                        let rankBadge = `<span class="font-bold text-slate-500 text-xs">${i+1}</span>`;
                        if(i===0) rankBadge = '🥇';
                        if(i===1) rankBadge = '🥈';
                        if(i===2) rankBadge = '🥉';

                        const userName = a.userName || 'Unknown';

                        rows += `
                        <div class="flex items-center p-3 border-b dark:border-dark-tertiary ${isMe?'bg-indigo-50 dark:bg-indigo-900/40 border-indigo-200':''}">
                            <div class="w-8 text-center">${rankBadge}</div>
                            <div class="flex-1 ml-3">
                                <div class="font-bold text-sm dark:text-white truncate">${userName} ${isMe?'(You)':''}</div>
                                <div class="text-[9px] text-slate-400"><i class="far fa-clock"></i> ${timeDisplay}</div>
                            </div>
                            <div class="font-bold text-indigo-600 dark:text-indigo-400 bg-indigo-50 dark:bg-indigo-900 px-2 py-1 rounded text-xs">${currentScore.toFixed(2)}</div>
                        </div>`;
                    });

                    const seeMoreBtn = window.leaderboardHasMore ? `
                        <div class="text-center mt-4">
                            <button onclick="Student.loadMoreRankings('${examId}', ${totalQuestions})" class="bg-indigo-500 text-white px-4 py-2 rounded-lg text-sm">See More</button>
                        </div>` : '';

                    const title = window.currentExamTitle || 'Exam';

                    c.innerHTML = renderHeader('rank') + `
                    <div class="p-5 pb-20">
                        <button onclick="Student.loadRankings()" class="mb-4 text-xs font-bold text-slate-500"><i class="fas fa-arrow-left"></i> ফিরে যান</button>
                        <div class="bg-gradient-to-r from-indigo-500 to-indigo-600 rounded-2xl p-6 text-white shadow-lg mb-6 text-center">
                            <h3 class="font-bold text-lg">${title}</h3>
                            <div class="flex flex-wrap justify-around mt-4 text-sm gap-2">
                                <div>অংশগ্রহণকারী: <b>${allAttempts.length}</b></div>
                                <div class="bg-white/20 px-3 py-0.5 rounded-full">আমার র‍্যাংক: <b>${myRank}</b></div>
                                <div>সর্বোচ্চ স্কোর: <b>${highestScore.toFixed(2)}</b></div>
                            </div>
                        </div>
                        <div class="bg-white dark:bg-dark-secondary rounded-xl border dark:border-dark-tertiary overflow-hidden shadow-sm">
                            ${rows || '<div class="p-10 text-center text-slate-400">এখনো কোনো রেজাল্ট নেই।</div>'}
                        </div>
                        ${seeMoreBtn}
                    </div>`;

                    if (!groupId) return [];

                    if (!groupDoc.exists()) return [];
                    
                    const groupData = groupDoc.data();
                    const studentIds = groupData.studentIds || [];
                    
                    if (studentIds.length === 0) return [];
                    
                    const students = [];
                    
                    for (const studentId of studentIds) {

                        if (studentDoc.exists()) {
                            const studentData = studentDoc.data();
                            if (studentData.blocked) continue;
                            
                            students.push({
                                id: studentId,
                                name: studentData.name || "No Name",
                                email: studentData.email || "No Email",
                                disabled: studentData.disabled || false
                            });
                        }
                    }
                    
                    return students;
                } catch (error) {
                    console.error("Error getting group students:", error);
                    return [];
                }

            loadLiveExams: async () => {
                const myRouteId = window.currentRouteId;
                if (AppState.userDisabled) {
                    Swal.fire('Access Restricted', 'Your account has been disabled. You cannot view exams.', 'warning');
                    return;
                }
                if (!AppState.activeGroupId) {
                    Swal.fire('Group Required', 'You need to join a group first', 'warning').then(() => Student.showGroupCodeModal());
                    return;
                }

                const c = document.getElementById('app-container');
                c.innerHTML = renderHeader('dashboard') + '<div class="p-10 text-center"><div class="loader mx-auto"></div><p class="text-xs text-slate-400 mt-2">Loading Live Exams...</p></div>';

                const uid = auth.currentUser ? auth.currentUser.uid : null;
                const now = new Date();

                const userAttemptsQuery = query(collection(db, "attempts"), 
                    where("userId", "==", uid),
                    where("isPractice", "==", false));
                    
                const userAttemptsSnap = await getDocs(userAttemptsQuery);
                const userAttempts = {};
                userAttemptsSnap.forEach(doc => {
                    const attempt = doc.data();
                    userAttempts[attempt.examId] = { id: doc.id, ...attempt };
                });

                const unsub = onSnapshot(query(collection(db, "exams"), where("groupId", "==", AppState.activeGroupId)), (snap) => {
                    if (myRouteId !== window.currentRouteId) return;

                    const ongoingExams = [];
                    const upcomingExams = [];

                    snap.forEach(doc => {
                        const e = { id: doc.id, ...doc.data() };
                        if (e.type === 'live' && !e.isDraft && !e.cancelled && !e.resultPublished) {
                            const startTime = new Date(e.startTime);
                            const endTime = new Date(e.endTime);

                            if (now >= startTime && now <= endTime) {
                                ongoingExams.push({...e, startTime, endTime, status: 'ongoing'});
                            } else if (now < startTime) {
                                upcomingExams.push({...e, startTime, endTime, status: 'upcoming'});
                            }
                        }
                    });

                    upcomingExams.sort((a, b) => a.startTime - b.startTime);
                    ongoingExams.sort((a, b) => a.endTime - b.endTime);

                    const renderLiveExamCard = (exam, status = 'ongoing') => {
                        const startTime = exam.startTime ? moment(exam.startTime).format('DD MMM, h:mm A') : '';
                        const endTime = exam.endTime ? moment(exam.endTime).format('h:mm A') : '';
                        const isAttended = userAttempts[exam.id] !== undefined;
                        
                        let buttonHTML = '';
                        let statusBadge = '';
                        let liveIndicator = '';
                        
                        if (status === 'ongoing') {
                            statusBadge = '<span class="live-status-badge status-live">Live Now</span>';
                            liveIndicator = '<div class="live-indicator live-now"></div>';
                            if (isAttended) {
                                buttonHTML = `
                                    <div class="mt-3">
                                        <button class="w-full bg-gray-300 dark:bg-gray-700 text-gray-600 dark:text-gray-400 py-2 rounded-lg text-sm font-bold cursor-not-allowed" disabled>
                                            <i class="fas fa-check-circle mr-2"></i> Already Attended
                                    </button>
                                </div>`;
                            } else {
                                buttonHTML = `
                                    <div class="mt-3">
                                        <button onclick="Exam.start('${exam.id}')" class="w-full bg-red-600 hover:bg-red-700 text-white py-2 rounded-lg text-sm font-bold">
                                            <i class="fas fa-play-circle mr-2"></i> Join Now
                                    </button>
                                </div>`;
                            }
                        } else if (status === 'upcoming') {
                            statusBadge = '<span class="live-status-badge status-upcoming">Upcoming</span>';
                            liveIndicator = '<div class="live-indicator upcoming"></div>';
                            buttonHTML = `
                                <div class="mt-3">
                                    <button class="w-full bg-blue-100 dark:bg-blue-900 text-blue-600 dark:text-blue-300 py-2 rounded-lg text-sm font-bold cursor-not-allowed" disabled>
                                        <i class="far fa-clock mr-2"></i> Starts at ${startTime}
                                    </button>
                                </div>`;
                        }
                        
                        return `
                        <div class="bg-white dark:bg-dark-secondary p-4 rounded-2xl shadow-sm border dark:border-dark-tertiary mb-3">
                            <div class="flex justify-between items-start mb-2">
                                <div>
                                    ${liveIndicator}
                                    ${statusBadge}
                                </div>
                                ${isAttended ? `<span class="text-[10px] font-bold bg-emerald-50 dark:bg-emerald-900 text-emerald-600 dark:text-emerald-300 px-2 py-1 rounded">Attended</span>` : ''}
                            </div>
                            <h3 class="font-bold text-lg text-slate-800 dark:text-white leading-tight mb-2 text-center">${exam.title}</h3>
                            ${exam.subject ? `<div class="text-xs text-indigo-500 dark:text-indigo-400 font-bold mb-2 text-center">${exam.subject} ${exam.chapter ? '• '+exam.chapter : ''}</div>` : ''}
                            <p class="text-xs text-slate-500 dark:text-slate-400 mb-3 flex gap-3 justify-center">
                                <span><i class="fas fa-star text-amber-400"></i> ${exam.totalMarks} Marks</span>
                                <span><i class="far fa-clock"></i> ${exam.duration}m</span>
                            </p>
                            <div class="text-xs text-slate-500 dark:text-slate-400 mb-3 text-center">
                                <i class="far fa-calendar"></i> ${startTime} - ${endTime}
                            </div>
                            ${buttonHTML}
                        </div>`;
                    };

                    let ongoingHTML = '';
                    if (ongoingExams.length > 0) {
                        ongoingHTML = `
                            <div class="mb-8">
                                <h3 class="text-lg font-bold mb-3 dark:text-white">Live Now</h3>
                                ${ongoingExams.map(exam => renderLiveExamCard(exam, 'ongoing')).join('')}
                            </div>
                        `;
                    }

                    let upcomingHTML = '';
                    if (upcomingExams.length > 0) {
                        upcomingHTML = `

            loadPastLiveExams: async () => {
                if (AppState.userDisabled) return;
                const c = document.getElementById('app-container');
                c.innerHTML = renderHeader('dashboard') + '<div class="p-10 text-center"><div class="loader mx-auto"></div><p class="text-xs text-slate-400 mt-2">Loading Past Exams...</p></div>';

                try {
                    const uid = auth.currentUser.uid;
                    const now = new Date();

                    const q = query(
                        collection(db, "exams"),
                        where("groupId", "==", AppState.activeGroupId),
                        where("type", "==", "live")
                    );
                    const snap = await getDocs(q);
                    const liveExams = [];
                    snap.forEach(doc => {
                        const e = { id: doc.id, ...doc.data() };
                        if (e.resultPublished || (e.endTime && new Date(e.endTime) < now) || (e.startTime && new Date(e.startTime) < now)) {
                            liveExams.push(e);
                        }
                    });

                    const userAttemptsSnap = await getDocs(query(
                        collection(db, "attempts"),
                        where("userId", "==", uid),
                        where("isPractice", "==", false)
                    ));
                    const userAttempts = {};
                    userAttemptsSnap.forEach(d => {
                        userAttempts[d.data().examId] = { id: d.id, ...d.data() };
                    });

                    const attendedExams = [];
                    const absentExams = [];
                    liveExams.forEach(exam => {
                        if (userAttempts[exam.id]) attendedExams.push({...exam, attempt: userAttempts[exam.id]});
                        else absentExams.push(exam);
                    });

                    // Extract unique subjects for filter buttons
                    const subjects = new Set();
                    liveExams.forEach(exam => {
                        const sub = exam.subject && exam.subject.trim() ? exam.subject.trim() : 'Uncategorized';
                        subjects.add(sub);
                    });
                    const subjectList = Array.from(subjects).sort();

                    const renderExamCard = (exam, isAttended = false) => {
                        const examDate = moment(exam.createdAt.toDate()).format('DD MMM, YYYY');
                        const startTime = exam.startTime ? moment(exam.startTime).format('h:mm A') : '';
                        // Show subject (if missing, show Uncategorized)
                        const subjectDisplay = exam.subject && exam.subject.trim() ? exam.subject : 'Uncategorized';
                        const chapterDisplay = exam.chapter ? ` • ${exam.chapter}` : '';
                        
                        let actions = `
                            <div class="action-grid">

                                    : `<button onclick="Student.viewExamSolutions('${exam.id}', 'live')" class="past-action-btn solutions bg-amber-500 text-white text-[10px] py-2 px-1 rounded shadow"> সমাধান </button>`}
                            </div>`;

                        <div class="bg-white dark:bg-dark-secondary p-4 rounded-2xl shadow-sm border dark:border-dark-tertiary mb-3">
                            <div class="flex justify-between items-start mb-2">
                                <span class="live-status-badge status-ended">Ended</span>
                                ${isAttended 
                                    ? `<span class="text-[10px] font-bold bg-emerald-50 dark:bg-emerald-900 text-emerald-600 dark:text-emerald-300 px-2 py-1 rounded">Attended</span>` 
                                    : `<span class="text-[10px] font-bold bg-amber-50 dark:bg-amber-900 text-amber-600 dark:text-amber-300 px-2 py-1 rounded">Absent</span>`}
                            </div>
                            <h3 class="font-bold text-lg text-slate-800 dark:text-white leading-tight mb-2 text-center">${exam.title}</h3>
                            <div class="text-xs text-indigo-500 dark:text-indigo-400 font-bold mb-2 text-center">${subjectDisplay}${chapterDisplay}</div>
                            <p class="text-xs text-slate-500 dark:text-slate-400 mb-3 flex gap-3 justify-center">
                                <span><i class="fas fa-star text-amber-400"></i> ${exam.totalMarks} Marks</span>
                                <span><i class="far fa-clock"></i> ${exam.duration}m</span>
                                <span><i class="far fa-calendar"></i> ${examDate}</span>
                            </p>
                            ${actions}
                        </div>`;
                    };

                    // Filter by subject
                    let filteredAttended = attendedExams;
                    let filteredAbsent = absentExams;
                    if (window.pastLiveExamSubjectFilter !== 'all') {
                        filteredAttended = attendedExams.filter(e => {
                            const sub = e.subject && e.subject.trim() ? e.subject.trim() : 'Uncategorized';
                            return sub === window.pastLiveExamSubjectFilter;
                        });
                        filteredAbsent = absentExams.filter(e => {
                            const sub = e.subject && e.subject.trim() ? e.subject.trim() : 'Uncategorized';
                            return sub === window.pastLiveExamSubjectFilter;
                        });
                    }

                    const allFiltered = [...filteredAttended, ...filteredAbsent];
                    allFiltered.sort((a, b) => (b.endTime ? new Date(b.endTime) : new Date(b.createdAt)) - (a.endTime ? new Date(a.endTime) : new Date(a.createdAt)));

            addTeacherCode: async () => {
                const { value: teacherCode } = await Swal.fire({
                    title: 'Add Teacher Account',
                    input: 'text',
                    inputLabel: 'Enter Teacher Code',
                    inputPlaceholder: 'Enter teacher code',
                    showCancelButton: true,
                    inputValidator: (value) => {
                        if (!value) return 'You need to enter a teacher code!';
                    }
                });
                
                if (teacherCode) {
                    try {
                        const isValidTeacherCode = await Student.verifyTeacherCode(teacherCode);
                        if (!isValidTeacherCode) {
                            Swal.fire('Error', 'Invalid teacher code. Please enter a valid teacher code.', 'error');
                            return;
                        }
                        
                        const user = auth.currentUser;
                        if (!user) return;
                        
                        const currentCodes = AppState.teacherCodes || [];
                        
                        const existingCode = currentCodes.find(tc => tc.code === teacherCode);
                        if (existingCode) {
                            Swal.fire('Error', 'This teacher code is already added', 'error');
                            return;
                        }
                        
                        const teacherInfo = await Student.getTeacherInfo(teacherCode);
                        
                        const newCodeObj = { code: teacherCode, active: currentCodes.length === 0 };
                        const updatedCodes = [...currentCodes, newCodeObj];
                        
                        await updateDoc(doc(db, "students", user.uid), { teacherCodes: updatedCodes });
                        
                        AppState.teacherCodes = updatedCodes;
                        AppState.teacherNames[teacherCode] = teacherInfo.name;
                        if (newCodeObj.active) {
                            AppState.activeTeacherCode = teacherCode;
                            
                            const teacherGroups = await Student.loadGroupsForCurrentTeacher();
                            if (teacherGroups.length > 0) {
                                AppState.activeGroupId = teacherGroups[0].groupId;
                                localStorage.setItem('activeGroupId', AppState.activeGroupId);
                                refreshExamCache();
                            }
                        }
                        
                        // Update cached profile
                        AppState.userProfile.teacherCodes = updatedCodes;
                        localStorage.setItem('userProfile', JSON.stringify(AppState.userProfile));
                        
                        Swal.fire('Success', 'Teacher account added successfully!', 'success').then(() => Student.loadManagement());
                    } catch (error) {
                        Swal.fire('Error', 'Failed to add teacher code: ' + error.message, 'error');
                    }
                }

            switchTeacherCode: async (codeToSwitch) => {
                try {
                    const user = auth.currentUser;
                    if (!user) return;
                    
                    const updatedCodes = AppState.teacherCodes.map(tc => ({ ...tc, active: tc.code === codeToSwitch }));
                    
                    await updateDoc(doc(db, "students", user.uid), { teacherCodes: updatedCodes });
                    
                    AppState.teacherCodes = updatedCodes;
                    AppState.activeTeacherCode = codeToSwitch;
                    
                    // Update cached profile
                    AppState.userProfile.teacherCodes = updatedCodes;
                    localStorage.setItem('userProfile', JSON.stringify(AppState.userProfile));
                    
                    const teacherGroups = await Student.loadGroupsForCurrentTeacher();
                    if (teacherGroups.length > 0) {
                        AppState.activeGroupId = teacherGroups[0].groupId;
                        localStorage.setItem('activeGroupId', AppState.activeGroupId);
                        refreshExamCache();
                    } else {
                        AppState.activeGroupId = null;
                        localStorage.removeItem('activeGroupId');
                        window.ExamCache = {};
                    }
                    
                    Swal.fire('Success', 'Switched to teacher account successfully!', 'success').then(() => Student.loadManagement());
                } catch (error) {
                    Swal.fire('Error', 'Failed to switch teacher account: ' + error.message, 'error');
                }

                        <p>Are you sure you want to delete this teacher account?</p>
                        <p class="text-sm text-red-500 mt-2">Type <strong>QuickZ</strong> below to confirm:</p>
                        <input type="text" id="delete-confirm" class="swal2-input mt-2" placeholder="Type QuickZ here">
                    `,
                    showCancelButton: true,
                    confirmButtonColor: '#d33',
                    confirmButtonText: 'Delete',
                    preConfirm: () => {
                        const confirmInput = document.getElementById('delete-confirm');
                        if (!confirmInput || confirmInput.value !== 'QuickZ') {
                            Swal.showValidationMessage('Please type "QuickZ" to confirm');
                            return false;
                        }
                        return true;
                    }
                });
                
                if (confirmText) await Student.removeTeacherCode(teacherCode);
            },
            
            removeTeacherCode: async (codeToRemove) => {
                if (AppState.teacherCodes.length <= 1) {
                    Swal.fire('Error', 'You must have at least one teacher account', 'error');
                    return;
                }
                
                try {
                    const user = auth.currentUser;
                    if (!user) return;
                    
                    const removedCode = AppState.teacherCodes.find(tc => tc.code === codeToRemove);
                    const updatedCodes = AppState.teacherCodes.filter(tc => tc.code !== codeToRemove);
                    
                    if (removedCode && removedCode.active) {
                        updatedCodes[0].active = true;
                        AppState.activeTeacherCode = updatedCodes[0].code;
                        
                        const teacherGroups = await Student.loadGroupsForCurrentTeacher();
                        if (teacherGroups.length > 0) {
                            AppState.activeGroupId = teacherGroups[0].groupId;
                            localStorage.setItem('activeGroupId', AppState.activeGroupId);
                            refreshExamCache();
                        } else {
                            AppState.activeGroupId = null;
                            localStorage.removeItem('activeGroupId');
                            window.ExamCache = {};
                        }
                    }
                    
                    await updateDoc(doc(db, "students", user.uid), { teacherCodes: updatedCodes });
                    
                    AppState.teacherCodes = updatedCodes;
                    
                    // Update cached profile
                    AppState.userProfile.teacherCodes = updatedCodes;
                    localStorage.setItem('userProfile', JSON.stringify(AppState.userProfile));
                    
                    Swal.fire('Success', 'Teacher account removed successfully', 'success').then(() => Student.loadManagement());
                } catch (error) {
                    Swal.fire('Error', 'Failed to remove teacher account: ' + error.message, 'error');
                }
            },
            
            leaveGroup: async (groupId) => {
                try {
                    const user = auth.currentUser;
                    if (!user) return;
                    
                    const updatedGroups = AppState.joinedGroups.filter(g => g.groupId !== groupId);
                    
                    await updateDoc(doc(db, "students", user.uid), { joinedGroups: updatedGroups });
                    
                    const groupDoc = await getDoc(doc(db, "groups", groupId));
                    if (groupDoc.exists()) {
                        const groupData = groupDoc.data();
                        const updatedStudentIds = (groupData.studentIds || []).filter(id => id !== user.uid);
                        
                        await updateDoc(doc(db, "groups", groupId), { studentIds: updatedStudentIds });
                    }
                    
                    AppState.joinedGroups = updatedGroups;
                    
                    if (AppState.activeGroupId === groupId) {
                        const teacherGroups = await Student.loadGroupsForCurrentTeacher();
                        if (teacherGroups.length > 0) {
                            AppState.activeGroupId = teacherGroups[0].groupId;
                            localStorage.setItem('activeGroupId', AppState.activeGroupId);
                        } else {
                            AppState.activeGroupId = null;
                            localStorage.removeItem('activeGroupId');
                        }
                        
                        refreshExamCache();
                    }
                    
                    // Update cached profile
                    AppState.userProfile.joinedGroups = updatedGroups;
                    localStorage.setItem('userProfile', JSON.stringify(AppState.userProfile));
                    
                    Swal.fire('Success', 'Left group successfully', 'success').then(() => Student.loadManagement());
                } catch (error) {
                    Swal.fire('Error', 'Failed to leave group: ' + error.message, 'error');
                }
            },
            
            changePassword: async () => {
                const currentPassword = document.getElementById('current-password').value;
                const newPassword = document.getElementById('new-password').value;
                const confirmPassword = document.getElementById('confirm-password').value;
                
                if (!currentPassword || !newPassword || !confirmPassword) {
                    Swal.fire('Error', 'Please fill all password fields', 'error');
                    return;
                }
                
                if (newPassword !== confirmPassword) {
                    Swal.fire('Error', 'New passwords do not match', 'error');
                    return;
                }
                
                if (newPassword.length < 6) {
                    Swal.fire('Error', 'New password must be at least 6 characters', 'error');
                    return;
                }
                
                try {
                    const user = auth.currentUser;
                    await signInWithEmailAndPassword(auth, user.email, currentPassword);
                    await updatePassword(user, newPassword);
                    
                    document.getElementById('current-password').value = '';
                    document.getElementById('new-password').value = '';
                    document.getElementById('confirm-password').value = '';
                    
                    Student.togglePasswordDropdown();
                    Swal.fire('Success', 'Password changed successfully!', 'success');
                } catch (error) {
                    if (error.code === 'auth/wrong-password') {
                        Swal.fire('Error', 'Current password is incorrect', 'error');
                    } else {
                        Swal.fire('Error', 'Failed to change password: ' + error.message, 'error');
                    }
                }
            }
        };

        // --- EXAM ENGINE ---
        window.Exam = {
            d: null, ans: [], marked: [], t: null, currentPage: 0, isPractice: false, startedAt: null,
            
            start: async (id, forcePractice = false) => {
                if (AppState.userDisabled) {
                    Swal.fire('Access Restricted', 'আপনার অ্যাকাউন্টটি ডিজেবল করা আছে।', 'warning');
                    return;
                }

                try {
                    const docSnap = await getDoc(doc(db, "exams", id));
                    if (!docSnap.exists()) {
                        Swal.fire('Error', 'পরীক্ষাটি পাওয়া যায়নি!', 'error');
                        return;
                    }
                    
                    const exam = { id: docSnap.id, ...docSnap.data() };
                    
                    // লাইভ এক্সাম চেকিং এ forcePractice যুক্ত করা হয়েছে
                    if (exam.type === 'live' && !forcePractice) {
                        const uid = auth.currentUser.uid;
                        const attemptQuery = query(
                            collection(db, "attempts"), 
                            where("userId", "==", uid), 
                            where("examId", "==", id)
                        );
                        const attemptSnap = await getDocs(attemptQuery);
                        
                        if (attemptSnap.size >= 2) {
                            Swal.fire('অ্যাক্সেস নেই!', 'আপনি ইতিমধ্যে সর্বোচ্চবার এই পরীক্ষায় প্রবেশ করেছেন।', 'error');
                            return;
                        }

                        try {
                            await addDoc(collection(db, "attempts"), {
                                userId: uid,
                                userName: AppState.userProfile?.name || auth.currentUser.displayName,
                                examId: id,
                                examTitle: exam.title,
                                status: 'locked',
                                entryNumber: attemptSnap.size + 1,
                                startedAt: new Date(),
                                score: 0,
                                isPractice: false,
                                groupId: AppState.activeGroupId
                            });
                        } catch (err) {
                            console.error("Lock error:", err);
                        }
                    }

                    try {
                        let questions;
                        try {
                            questions = JSON.parse(exam.questions);
                        } catch (parseError) {
                            console.error("Question parse error:", parseError);
                            Swal.fire('Error', 'প্রশ্নপত্রে ত্রুটি আছে (JSON Error)', 'error');
                            return;
                        }
                        
                        // Remove correct answers from questions for client-side
                        const questionsWithoutCorrect = questions.map(q => {
                            const { correct, ...rest } = q;
                            return rest;
                        });
                        
                        Exam.d = { ...exam, qs: questionsWithoutCorrect };
                        Exam.ans = new Array(Exam.d.qs.length).fill(null);
                        Exam.marked = new Array(Exam.d.qs.length).fill(false);
                        Exam.currentPage = 0;
                        Exam.startedAt = new Date();
                        
                        // Mock হিসেবে মার্ক করার লজিক
                        Exam.isPractice = forcePractice || (exam.type === 'mock');
                        
                        document.getElementById('review-panel-btn').classList.remove('hidden');
                        
                        if (exam.type === 'live' && !forcePractice) Swal.close();
                        
                        await Exam.render();
                        Exam.runT(Exam.d.duration * 60);
                        Exam.updateReviewPanel();
                    } catch (error) {
                        if (exam.type === 'live' && !forcePractice) Swal.close();
                        console.error("Exam start error:", error);
                        Swal.fire('Error', 'পরীক্ষা শুরু করতে সমস্যা: ' + error.message, 'error');
                    }
                } catch (error) {
                    Swal.fire('Error', 'পরীক্ষা লোড করতে সমস্যা: ' + error.message, 'error');
                }
            },
            
            toggleMark: (index) => {
                Exam.marked[index] = !Exam.marked[index];
                const btn = document.getElementById(`mark-btn-${index}`);
                if (btn) {
                    if (Exam.marked[index]) {
                        btn.classList.add('text-amber-500');
                        btn.classList.remove('text-slate-400');
                        btn.innerHTML = '<i class="fas fa-bookmark"></i> Marked';
                    } else {
                        btn.classList.remove('text-amber-500');
                        btn.classList.add('text-slate-400');
                        btn.innerHTML = '<i class="far fa-bookmark"></i> Mark for review';
                    }
                }
                Exam.updateReviewPanel();
            },
            
            updateReviewPanel: () => {
                const panel = document.getElementById('question-numbers');
                if (!panel) return;
                panel.innerHTML = '';
                const questionsPerPage = 25;

                Exam.d.qs.forEach((_, i) => {
                    const btn = document.createElement('button');
                    btn.className = 'question-number-btn';
                    if (Exam.ans[i] !== null) btn.classList.add('answered');
                    
                    const startIdx = Exam.currentPage * questionsPerPage;
                    const endIdx = startIdx + questionsPerPage;
                    if (i >= startIdx && i < endIdx) btn.classList.add('current-view');
                    
                    btn.textContent = i + 1;
                    btn.onclick = async () => {
                        const targetPage = Math.floor(i / questionsPerPage);
                        if (targetPage !== Exam.currentPage) {
                            Exam.currentPage = targetPage;
                            await Exam.render();
                        }
                        setTimeout(() => {
                            const el = document.getElementById(`q-${i}`);
                            if (el) {
                                el.scrollIntoView({ behavior: 'instant', block: 'center' });
                                el.style.backgroundColor = 'rgba(79, 70, 229, 0.1)';
                                el.style.transition = 'background-color 0.3s';
                                setTimeout(() => el.style.backgroundColor = '', 1000);
                            }
                        }, 50);
                        document.getElementById('review-panel').classList.remove('show');
                    };
                    panel.appendChild(btn);
                });
            },
            
            render: async () => {
                const questionsPerPage = 25;
                const startIndex = Exam.currentPage * questionsPerPage;
                const endIndex = Math.min(startIndex + questionsPerPage, Exam.d.qs.length);
                
                const answeredCount = Exam.ans.filter(a => a !== null).length;
                
                let questionHTML = '';
                for (let i = startIndex; i < endIndex; i++) {
                    const q = Exam.d.qs[i];
                    
                    const qText = window.MathHelper.renderExamContent(q.q);
                    
                    const isAnswered = Exam.ans[i] !== null;
                    
                    questionHTML += `<div class="bg-white dark:bg-dark-secondary p-4 rounded-xl shadow-sm border dark:border-dark-tertiary mb-4" id="q-${i}">
                        <div class="flex justify-between items-center mb-3">
                            <span class="bg-indigo-50 dark:bg-indigo-900 text-indigo-600 dark:text-indigo-300 px-2 py-0.5 text-sm rounded h-fit">${i+1}</span>
                            <button id="mark-btn-${i}" onclick="Exam.toggleMark(${i})" class="text-xs font-bold ${Exam.marked[i] ? 'text-amber-500' : 'text-slate-400'}">
                                <i class="${Exam.marked[i] ? 'fas' : 'far'} fa-bookmark"></i> ${Exam.marked[i] ? 'Marked' : 'Mark for review'}
                            </button>
                        </div>
                        <p class="font-bold mb-3 text-slate-800 dark:text-white text-left flex gap-2">
                            <span class="flex-1" style="font-family: 'SolaimanLipi', sans-serif; font-size: 14px; line-height: 1.5;">${qText}</span>
                        </p>
                        <div class="space-y-2">
                            ${q.options.map((o,oi) => {
                                const selected = Exam.ans[i] === oi ? 'selected' : '';
                                const locked = isAnswered ? 'locked' : '';
                                
                                const optionText = window.MathHelper.renderExamContent(o);
                                
                                return `<button onclick="Exam.sel(${i},${oi})" class="opt-btn option-btn w-full text-left p-3 rounded-lg border text-sm flex gap-2 transition ${selected} ${locked}" ${isAnswered ? 'disabled' : ''}>
                                    <span class="font-bold opacity-50 w-6">${String.fromCharCode(65+oi)}.</span>
                                    <span class="flex-1 text-left" style="font-family: 'SolaimanLipi', sans-serif; font-size: 14px; line-height: 1.5;">${optionText}</span>
                                    ${selected ? '<i class="fas fa-check text-indigo-600 ml-2"></i>' : ''}
                                </button>`;
                            }).join('')}
                        </div>
                    </div>`;
                }
                
                const totalPages = Math.ceil(Exam.d.qs.length / questionsPerPage);
                let paginationHTML = '';
                if (totalPages > 1) {
                    paginationHTML = `
                        <div class="flex justify-center gap-2 mt-6">
                            <button onclick="Exam.prevPage()" class="px-4 py-2 bg-slate-100 dark:bg-dark-tertiary rounded-lg ${Exam.currentPage === 0 ? 'opacity-50 cursor-not-allowed' : ''}" ${Exam.currentPage === 0 ? 'disabled' : ''}>
                                <i class="fas fa-chevron-left"></i> Previous
                            </button>
                            <span class="px-4 py-2 text-sm">Page ${Exam.currentPage + 1} of ${totalPages}</span>
                            <button onclick="Exam.nextPage()" class="px-4 py-2 bg-slate-100 dark:bg-dark-tertiary rounded-lg ${Exam.currentPage === totalPages - 1 ? 'opacity-50 cursor-not-allowed' : ''}" ${Exam.currentPage === totalPages - 1 ? 'disabled' : ''}>
                                Next <i class="fas fa-chevron-right"></i>
                            </button>
                        </div>
                    `;
                }
                
                const practiceIndicator = Exam.isPractice ? 
                    `<div class="text-xs bg-amber-100 dark:bg-amber-900 text-amber-800 dark:text-amber-200 px-2 py-1 rounded inline-block mb-2 text-center">
                        <i class="fas fa-flask"></i> Practice Mode
                    </div>` : '';
                
                document.getElementById('app-container').innerHTML = `
                <div class="sticky top-0 bg-white dark:bg-dark-secondary border-b dark:border-dark-tertiary px-4 py-3 flex justify-between items-center z-30 shadow-md">
                    <div>
                        ${practiceIndicator}
                        <div class="text-center">
                            <span class="font-bold block text-sm truncate w-32 dark:text-white mx-auto">${Exam.d.title}</span>
                            <div class="flex items-center justify-center gap-2 mt-1">
                                <span id="tm" class="text-xs font-mono bg-slate-100 dark:bg-dark-tertiary px-1 rounded text-slate-600 dark:text-slate-300">00:00</span>
                                <span id="answered-counter" class="text-xs font-mono bg-indigo-100 dark:bg-indigo-900 text-indigo-600 dark:text-indigo-300 px-1 rounded">সম্পন্ন: ${answeredCount}/${Exam.d.qs.length}</span>
                            </div>
                        </div>
                    </div>
                    <div class="flex gap-2">
                        <button onclick="Exam.sub()" class="bg-gradient-to-r from-indigo-500 to-indigo-600 text-white px-5 py-2 rounded-lg text-sm font-bold shadow hover:bg-indigo-700 transition">SUBMIT</button>
                    </div>
                </div>
                <div class="p-4 pb-20 bg-slate-50 dark:bg-black min-h-screen select-none exam-question-container">
                    ${questionHTML}
                    ${paginationHTML}
                </div>`;
                
                loadMathJax(null, document.getElementById('app-container'));
                Exam.updateReviewPanel();
            },
            
            sel: (qi, oi) => {
                if (Exam.ans[qi] !== null) return;
                
                Exam.ans[qi] = oi;
                
                const questionDiv = document.getElementById(`q-${qi}`);
                if (questionDiv) {
                    const buttons = questionDiv.querySelectorAll('.option-btn');
                    buttons.forEach((btn, index) => {
                        if (index === oi) {
                            btn.classList.add('selected');
                            if (!btn.querySelector('.fa-check')) {
                                const checkIcon = document.createElement('i');
                                checkIcon.className = 'fas fa-check text-indigo-600 ml-2';
                                btn.appendChild(checkIcon);
                            }
                        }
                        btn.classList.add('locked');
                        btn.disabled = true;
                    });
                }
                
                const answeredCount = Exam.ans.filter(a => a !== null).length;
                const counterEl = document.getElementById('answered-counter');
                if (counterEl) counterEl.textContent = `সম্পন্ন: ${answeredCount}/${Exam.d.qs.length}`;
                
                Exam.updateReviewPanel();
            },
            
            prevPage: () => {
                if (Exam.currentPage > 0) {
                    Exam.currentPage--;
                    Exam.render();
                    // Jump to top instantly
                    const container = document.getElementById('app-container');
                    container.scrollTop = 0;
                }
            },
            
            nextPage: () => {
                const totalPages = Math.ceil(Exam.d.qs.length / 25);
                if (Exam.currentPage < totalPages - 1) {
                    Exam.currentPage++;
                    Exam.render();
                    // Jump to top instantly
                    const container = document.getElementById('app-container');
                    container.scrollTop = 0;
                }
            },
            
            runT: (sec) => {
                Exam.t = setInterval(()=>{ 
                    sec--; 
                    document.getElementById('tm').innerText=`${Math.floor(sec/60)}:${(sec%60).toString().padStart(2,'0')}`; 
                    if(sec<=0) Exam.sub(true); 
                },1000);
            },
            
            sub: async (auto=false) => {
                if(!auto && !confirm('Submit Exam?')) return;
                clearInterval(Exam.t);
                
                const examDoc = await getDoc(doc(db, "exams", Exam.d.id));
                if (!examDoc.exists()) {
                    Swal.fire('Error', 'পরীক্ষার তথ্য পাওয়া যায়নি', 'error');
                    return;
                }
                const fullExam = examDoc.data();
                const fullQs = JSON.parse(fullExam.questions);
                
                const neg = fullExam.negativeMark ? parseFloat(fullExam.negativeMark) : 0;
                let score = 0;
                fullQs.forEach((q,i) => { 
                    if(Exam.ans[i] == q.correct) score++; 
                    else if(Exam.ans[i] !== null) score -= neg; 
                });
                score = Math.max(0, score);
                const showInstant = (fullExam.type === 'mock' || fullExam.resultPublished);
                
                await addDoc(collection(db, "attempts"), {
                    userId: auth.currentUser.uid, 
                    userName: AppState.userProfile?.name || auth.currentUser.displayName,
                    examId: Exam.d.id, 
                    examTitle: Exam.d.title, 
                    score, 
                    answers: Exam.ans,
                    markedAnswers: Exam.marked, // save marked array
                    startedAt: Exam.startedAt,
                    submittedAt: new Date(),
                    isPractice: Exam.isPractice || fullExam.type === 'mock'
                });
                
                if(showInstant) {
                    Swal.fire('Result', `Score: ${score}`, 'success').then(() => { 
                        document.getElementById('review-panel-btn').classList.add('hidden');
                        document.getElementById('review-panel').classList.remove('show');
                        
                        if (fullExam.type === 'mock') {
                            window.resultTypeFilter = 'mock';
                            Router.student('results');
                        } else {
                            Router.student('results');
                        }
                    });
                } else {
                    Swal.fire('Submitted', 'Wait for result.', 'success').then(() => {
                        document.getElementById('review-panel-btn').classList.add('hidden');
                        document.getElementById('review-panel').classList.remove('show');
                        Router.student('dashboard');
                    });
                }
            }
        };

        // Setup review panel button
        document.getElementById('review-panel-btn').addEventListener('click', function() {
            document.getElementById('review-panel').classList.toggle('show');
        });

        // Firestore subscription refresh
        const refreshExamCache = () => {
            clearListeners();
            if (!AppState.activeGroupId) {
                window.ExamCache = {};
                return;
            }
            
            const q = query(collection(db, "exams"), where("groupId", "==", AppState.activeGroupId));
            const unsubscribe = onSnapshot(q, (snap) => {
                window.ExamCache = {};
                snap.forEach(d => window.ExamCache[d.id] = { id: d.id, ...d.data() });
                console.log("Exam Data Synced");
                
                const currentPath = document.querySelector('.nav-links button.active')?.getAttribute('onclick');
                if (currentPath && currentPath.includes('dashboard')) Student.loadDashboard();
                if (currentPath && currentPath.includes('results')) Student.loadResults();
                if (currentPath && currentPath.includes('rank')) Student.loadRankings();
            });
            window.unsubscribes.push(unsubscribe);
        };

        // Close review panel when clicking outside
        document.addEventListener('click', function(e) {
            if (!e.target.closest('#review-panel') && !e.target.closest('#review-panel-btn')) {
                document.getElementById('review-panel').classList.remove('show');
            }
        });

        window.addEventListener('unhandledrejection', function(event) {
            console.error('Unhandled rejection:', event.reason);
            Swal.fire({
                icon: 'error',
                title: 'Something went wrong',
                text: event.reason.message || 'An unexpected error occurred',
                timer: 3000,
                showConfirmButton: false
            });
        });

        // দ্রুত লগইন অবস্থা চেক করার জন্য localStorage ব্যবহার
        const cachedUserLoggedIn = localStorage.getItem('userLoggedIn') === 'true';
        if (cachedUserLoggedIn) {
            // যদি ক্যাশে থাকে, তাহলে স্প্ল্যাশ স্ক্রিন লুকিয়ে অ্যাপ কন্টেইনার দেখান (লোডিং অবস্থায়)
            document.getElementById('splash-screen').classList.add('hidden');
            document.getElementById('app-container').classList.remove('hidden');
            // কিন্তু অ্যাপ রেন্ডার হবে না, কারণ অথ স্টেট চেঞ্জ হওয়ার পরে রেন্ডার হবে
        }

        onAuthStateChanged(auth, async (u) => {
            if (!u) { 
                AuthUI.showAuthScreen();
                document.getElementById('splash-screen').classList.add('hidden');
                localStorage.removeItem('userLoggedIn');
                return;
            }
            
            try {
                const cached = localStorage.getItem('userProfile');
                if (cached) {
                    const userData = JSON.parse(cached);
                    if (!userData.blocked) {
                        AppState.user = u;
                        AppState.userDisabled = userData.disabled || false;
                        AppState.profileCompleted = userData.profileCompleted || false;
                        AppState.userProfile = userData;
                        AppState.teacherCodes = userData.teacherCodes || [];
                        AppState.activeTeacherCode = AppState.teacherCodes.find(tc => tc.active)?.code || null;
                        AppState.joinedGroups = userData.joinedGroups || [];
                        
                        const storedGroupId = localStorage.getItem('activeGroupId');
                        if (storedGroupId && AppState.joinedGroups.find(g => g.groupId === storedGroupId)) {
                            AppState.activeGroupId = storedGroupId;
                        } else if (AppState.joinedGroups.length > 0) {
                            AppState.activeGroupId = AppState.joinedGroups[0].groupId;
                            localStorage.setItem('activeGroupId', AppState.activeGroupId);
                        }

                        if (AppState.teacherCodes.length > 0) await Student.loadTeacherNames();

                        if (!AppState.profileCompleted) Router.showProfileForm();
                        else Router.initStudent();
                        
                        document.getElementById('splash-screen').classList.add('hidden');
                        return;
                    }
                }
                
                const s = await getDoc(doc(db, "students", u.uid));
                if(s.exists()) {
                    const userData = s.data();
                    if(userData.blocked) {
                        await signOut(auth);
                        Swal.fire('Blocked', 'Your account is blocked.', 'error');
                        AuthUI.showAuthScreen();
                    } else {
                        AppState.user = u;
                        AppState.userDisabled = userData.disabled || false;
                        AppState.profileCompleted = userData.profileCompleted || false;
                        AppState.userProfile = userData;
                        AppState.teacherCodes = userData.teacherCodes || [];
                        AppState.activeTeacherCode = AppState.teacherCodes.find(tc => tc.active)?.code || null;
                        AppState.joinedGroups = userData.joinedGroups || [];
                        
                        localStorage.setItem('userProfile', JSON.stringify(userData));
                        localStorage.setItem('userLoggedIn', 'true');
                        
                        const storedGroupId = localStorage.getItem('activeGroupId');
                        if (storedGroupId && AppState.joinedGroups.find(g => g.groupId === storedGroupId)) {
                            AppState.activeGroupId = storedGroupId;
                        } else if (AppState.joinedGroups.length > 0) {
                            AppState.activeGroupId = AppState.joinedGroups[0].groupId;
                            localStorage.setItem('activeGroupId', AppState.activeGroupId);
                        }

                        if (AppState.teacherCodes.length > 0) await Student.loadTeacherNames();

                        if (!AppState.profileCompleted) Router.showProfileForm();
                        else Router.initStudent();
                    }
                } else {
                    await setDoc(doc(db, "students", u.uid), {
                        uid: u.uid,
                        email: u.email,
                        name: "",
                        phone: "",
                        fatherPhone: "",
                        motherPhone: "",
                        schoolName: "",
                        collegeName: "",
                        teacherCodes: [],
                        groupCode: "",
                        joinedGroups: [],
                        profileCompleted: false,
                        blocked: false,
                        disabled: false,
                        joined: new Date()
                    });
                    
                    AppState.profileCompleted = false;
                    Router.showProfileForm();
                }
            } catch (error) {
                console.error("Auth Error:", error);
                AuthUI.showAuthScreen();
            }
            
            document.getElementById('splash-screen').classList.add('hidden');
        });

        window.addEventListener('load', () => {
            setTimeout(() => {
                if (document.getElementById('splash-screen').classList.contains('hidden') === false) {
                    console.log("Still showing splash screen, showing auth...");
                    document.getElementById('splash-screen').classList.add('hidden');
                    AuthUI.showAuthScreen();
                }
            }, 3000);
        });
    </script>
