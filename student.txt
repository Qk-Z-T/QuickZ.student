            // Verify teacher code exists in Firebase
            verifyTeacherCode: async (teacherCode) => {
                try {
                    const teachersQuery = query(collection(db, "teachers"), where("teacherCode", "==", teacherCode));
                    const querySnapshot = await getDocs(teachersQuery);
                    return !querySnapshot.empty;

                    console.error("Error verifying teacher code:", error);

            // Get group information
            getGroupInfo: async (groupId) => {
                try {
                    if (!groupId) return { name: "No Group", totalStudents: 0 };
                    
                    const groupDoc = await getDoc(doc(db, "groups", groupId));
                    
                    if (!groupDoc.exists()) {
                        return { name: "Unknown Group", totalStudents: 0 };
                    }
                    
                    const groupData = groupDoc.data();
                    const studentIds = groupData.studentIds || [];
                    
                    return { 
                        name: groupData.name || groupData.code || "Unknown Group",
                        totalStudents: studentIds.length,
                        teacherName: groupData.teacherName || "",
                        teacherPhone: groupData.teacherPhone || "",
                        teacherCode: groupData.teacherCode || "",
                        teacherId: groupData.teacherId || ""
                    };
                } catch (error) {
                    console.error("Error getting group info:", error);
                    return { name: "Error Loading", totalStudents: 0 };
                }
            },

            // Get students in a group
            getGroupStudents: async (groupId) => {
                try {
                    if (!groupId) return [];
                    
                    const groupDoc = await getDoc(doc(db, "groups", groupId));
                    if (!groupDoc.exists()) return [];
                    
                    const groupData = groupDoc.data();
                    const studentIds = groupData.studentIds || [];
                    
                    if (studentIds.length === 0) return [];
                    
                    const students = [];
                    
                    // Fetch each student's data
                    for (const studentId of studentIds) {
                        const studentDoc = await getDoc(doc(db, "students", studentId));
                        if (studentDoc.exists()) {
                            const studentData = studentDoc.data();
                            // Skip blocked students
                            if (studentData.blocked) continue;
                            
                            students.push({
                                id: studentId,
                                name: studentData.name || "No Name",
                                email: studentData.email || "No Email",
                                disabled: studentData.disabled || false
                            });
                        }
                    }
                    
                    return students;
                } catch (error) {
                    console.error("Error getting group students:", error);
                    return [];
                }
            },

            saveGroupCode: async () => {
                const groupCode = document.getElementById('group-code-input').value.trim();
                
                if (!groupCode) {
                    Swal.fire('Error', 'গ্রুপ কোড দিন', 'error');
                    return;
                }
                
                try {
                    // ১. গ্রুপটি আদৌ আছে কি না চেক করা
                    const gQuery = query(collection(db, "groups"), where("groupCode", "==", groupCode));
                    const gSnap = await getDocs(gQuery);
                    
                    if (gSnap.empty) {
                        Swal.fire('Error', 'ভুল গ্রুপ কোড! এই কোডে কোনো গ্রুপ নেই।', 'error');
                        return;
                    }
                    
                    const groupDoc = gSnap.docs[0];
                    const groupData = groupDoc.data();
                    const groupId = groupDoc.id;
                    
                    // ২. গ্রুপটি স্টুডেন্টের একটিভ টিচারের কি না চেক করা
                    if (groupData.teacherCode !== AppState.activeTeacherCode) {
                        Swal.fire('Error', 'এই গ্রুপটি আপনার বর্তমান টিচারের নয়। টিচার কোড পরিবর্তন করুন অথবা সঠিক গ্রুপ কোড দিন।', 'error');
                        return;
                    }
                    
                    const user = auth.currentUser;
                    
                    // ৩. স্টুডেন্টের joinedGroups অ্যারেতে যুক্ত করা
                    let joinedGroups = AppState.userProfile.joinedGroups || [];

                        Swal.fire('Info', 'আপনি ইতিমধ্যে এই গ্রুপে যুক্ত আছেন', 'info');
                        return;

                    joinedGroups.push({ 
                        groupId, 
                        groupName: groupData.name
                    });
                    
                    await updateDoc(doc(db, "students", user.uid), { 
                        joinedGroups 
                    });
                    
                    // টিচার প্যানেলের স্টুডেন্ট লিস্ট আপডেট করার জন্য studentIds এ যুক্ত করা
                    const studentIds = groupData.studentIds || [];
                    if(!studentIds.includes(user.uid)) {
                        studentIds.push(user.uid);

                            studentIds 
                        });

                            const startTime = new Date(e.startTime);
                            const endTime = new Date(e.endTime);

                            if (now >= startTime && now <= endTime) {
                                ongoingExams.push({...e, startTime, endTime, status: 'ongoing'});

                                upcomingExams.push({...e, startTime, endTime, status: 'upcoming'});

                            statusBadge = '<span class="live-status-badge status-upcoming">Upcoming</span>';
                            buttonHTML = `
                                <div class="mt-3">
                                    <button class="w-full bg-blue-100 dark:bg-blue-900 text-blue-600 dark:text-blue-300 py-2 rounded-lg text-sm font-bold cursor-not-allowed" disabled>

                                </div>`;

                        <div class="bg-white dark:bg-dark-secondary p-4 rounded-2xl shadow-sm border dark:border-dark-tertiary mb-3">

                                ${statusBadge}

                                <span><i class="fas fa-star text-amber-400"></i> ${exam.totalMarks} Marks</span>
                                <span><i class="far fa-clock"></i> ${exam.duration}m</span>
                            </p>
                            <div class="text-xs text-slate-500 dark:text-slate-400 mb-3 text-center">
                                <i class="far fa-calendar"></i> ${startTime} - ${endTime}
                            </div>
                            ${buttonHTML}
                        </div>`;
                    };

                    let ongoingHTML = '';

                        ongoingHTML = `

                        const endTime = new Date(e.endTime);
                        return endTime < now;

                    if (e.startTime) {
                        const startTime = new Date(e.startTime);
                        return startTime < now;

            removeTeacherCode: async (codeToRemove) => {
                // Check if it's the last code
                if (AppState.teacherCodes.length <= 1) {
                    Swal.fire('Error', 'You must have at least one teacher account', 'error');
                    return;
                }
                
                const result = await Swal.fire({
                    title: 'Remove Teacher Account?',
                    text: "Are you sure you want to remove this teacher account?",
                    icon: 'warning',
                    showCancelButton: true,
                    confirmButtonColor: '#d33',
                    confirmButtonText: 'Remove'
                });
                
                if (result.isConfirmed) {
                    try {
                        const user = auth.currentUser;
                        if (!user) return;
                        
                        const removedCode = AppState.teacherCodes.find(tc => tc.code === codeToRemove);
                        const updatedCodes = AppState.teacherCodes.filter(tc => tc.code !== codeToRemove);
                        
                        // If we removed the active code, make the first one active
                        if (removedCode && removedCode.active) {
                            updatedCodes[0].active = true;
                            AppState.activeTeacherCode = updatedCodes[0].code;
                        }
                        
                        // Update in Firestore
                        await updateDoc(doc(db, "students", user.uid), {
                            teacherCodes: updatedCodes
                        });
                        
                        // Update local state
                        AppState.teacherCodes = updatedCodes;
                        
                        Swal.fire('Success', 'Teacher account removed successfully', 'success').then(() => {
                            Student.profile();
                        });
                    } catch (error) {
                        Swal.fire('Error', 'Failed to remove teacher account: ' + error.message, 'error');

            leaveGroup: async (groupId) => {
                const result = await Swal.fire({
                    title: 'Leave Group?',
                    text: "Are you sure you want to leave this group?",
                    icon: 'warning',
                    showCancelButton: true,
                    confirmButtonColor: '#d33',
                    confirmButtonText: 'Leave'
                });
                
                if (result.isConfirmed) {
                    try {
                        const user = auth.currentUser;
                        if (!user) return;
                        
                        // Remove from joinedGroups
                        const updatedGroups = AppState.joinedGroups.filter(g => g.groupId !== groupId);
                        
                        // Update student document
                        await updateDoc(doc(db, "students", user.uid), {
                            joinedGroups: updatedGroups
                        });
                        
                        // Remove from group's studentIds
                        const groupDoc = await getDoc(doc(db, "groups", groupId));
                        if (groupDoc.exists()) {
                            const groupData = groupDoc.data();
                            const updatedStudentIds = (groupData.studentIds || []).filter(id => id !== user.uid);
                            
                            await updateDoc(doc(db, "groups", groupId), {
                                studentIds: updatedStudentIds
                            });
                        }
                        
                        // Update local state
                        AppState.joinedGroups = updatedGroups;
                        
                        // If leaving active group, set new active group
                        if (AppState.activeGroupId === groupId) {
                            if (updatedGroups.length > 0) {
                                AppState.activeGroupId = updatedGroups[0].groupId;
                                localStorage.setItem('activeGroupId', AppState.activeGroupId);
                            } else {
                                AppState.activeGroupId = null;
                                localStorage.removeItem('activeGroupId');
                            }
                            
                            // Refresh exam cache for new active group
                            refreshExamCache();
                        }
                        
                        Swal.fire('Success', 'Left group successfully', 'success').then(() => {
                            Student.profile();
                        });
                    } catch (error) {
                        Swal.fire('Error', 'Failed to leave group: ' + error.message, 'error');
                    }
                }
            },

                        Swal.fire('Error', 'Failed to change password: ' + error.message, 'error');
                    }
                }
            }
        };

        // --- EXAM ENGINE ---
        window.Exam = {
            d: null, ans: [], t: null, currentPage: 0, isPractice: false, startedAt: null,
            
            // ৩. রিভিউ প্যানেল আপডেট এবং জাম্প ফিচার
            updateReviewPanel: () => {
                const panel = document.getElementById('question-numbers');
                if (!panel) return;
                panel.innerHTML = '';
                const questionsPerPage = 25;

                Exam.d.qs.forEach((_, i) => {
                    const btn = document.createElement('button');
                    btn.className = 'question-number-btn';
                    if (Exam.ans[i] !== null) btn.classList.add('answered');
                    
                    const startIdx = Exam.currentPage * questionsPerPage;
                    const endIdx = startIdx + questionsPerPage;
                    if (i >= startIdx && i < endIdx) btn.classList.add('current-view');
                    
                    btn.textContent = i + 1;
                    btn.onclick = async () => {
                        const targetPage = Math.floor(i / questionsPerPage);
                        if (targetPage !== Exam.currentPage) {
                            Exam.currentPage = targetPage;
                            await Exam.render();
                        }
                        setTimeout(() => {
                            const el = document.getElementById(`q-${i}`);
                            if (el) el.scrollIntoView({ behavior: 'smooth', block: 'center' });
                        }, 100);
                        document.getElementById('review-panel').classList.remove('show');
                    };
                    panel.appendChild(btn);
                });
            },
            
            // ১. ওয়ান-টাইম এন্ট্রি লজিক সহ এক্সাম স্টার্ট
            start: async (id) => {
                if (AppState.userDisabled) {
                    Swal.fire('Access Restricted', 'Your account has been disabled. You cannot view exams.', 'warning');
                    return;
                }

                // Check if has active group
                if (!AppState.activeGroupId) {
                    Swal.fire('Group Required', 'You need to join a group to take exams', 'warning').then(() => {
                        Student.showGroupCodeModal();
                    });
                    return;
                }

                const e = window.ExamCache[id];
                if(!e) { Swal.fire('Error', 'Exam Not Found', 'error'); return; }
                
                // Check if exam belongs to active group
                if (e.groupId !== AppState.activeGroupId) {
                    Swal.fire('Access Denied', 'This exam does not belong to your active group', 'error');
                    return;
                }
                
                // ১ বার এন্ট্রি লিমিট লজিক (Live Exam এর জন্য)
                if(e.type === 'live') {
                    const uid = auth.currentUser.uid;
                    
                    // চেক করা হচ্ছে আগে কোনো attempt আছে কি না
                    const attemptQuery = query(collection(db, "attempts"), where("userId", "==", uid), where("examId", "==", id));
                    const attemptSnap = await getDocs(attemptQuery);
                    
                    if (!attemptSnap.empty) {
                        Swal.fire('অ্যাক্সেস নেই!', 'আপনি ইতিমধ্যে এই পরীক্ষায় প্রবেশ করেছিলেন। পুনরায় প্রবেশের সুযোগ নেই।', 'error');
                        return;
                    }

                    // প্রবেশের সাথে সাথেই এন্ট্রি ব্লক করে দিন
                    await addDoc(collection(db, "attempts"), {
                        userId: uid,
                        examId: id,
                        status: 'locked',
                        startedAt: new Date(),
                        score: 0,
                        isPractice: false
                    });
                }

                Exam.d = { ...e, qs: JSON.parse(e.questions) };
                Exam.ans = new Array(Exam.d.qs.length).fill(null);
                Exam.currentPage = 0;
                Exam.startedAt = new Date();
                Exam.isPractice = (e.type === 'mock');
                
                document.getElementById('student-nav').classList.add('hidden');
                document.getElementById('review-panel-btn').classList.remove('hidden');
                
                await Exam.render();
                Exam.runT(Exam.d.duration * 60);
                Exam.updateReviewPanel();
            },
            
            // ২. বামমুখী এলাইনমেন্ট সহ প্রশ্ন রেন্ডার
            render: async () => {
                const questionsPerPage = 25;
                const startIndex = Exam.currentPage * questionsPerPage;
                const endIndex = Math.min(startIndex + questionsPerPage, Exam.d.qs.length);
                
                let questionHTML = '';
                for (let i = startIndex; i < endIndex; i++) {
                    const q = Exam.d.qs[i];
                    let qText = window.MathHelper.isMathContent(q.q) ? window.MathHelper.renderMath(q.q) : q.q;
                    
                    questionHTML += `<div class="bg-white dark:bg-dark-secondary p-4 rounded-xl shadow-sm border dark:border-dark-tertiary mb-4" id="q-${i}">
                        <p class="font-bold mb-3 text-slate-800 dark:text-white text-left flex gap-2">
                            <span class="bg-indigo-50 dark:bg-indigo-900 text-indigo-600 dark:text-indigo-300 px-2 py-0.5 text-sm rounded h-fit">${i+1}</span>
                            <span class="flex-1">${qText}</span>
                        </p>
                        <div class="space-y-2">
                            ${q.options.map((o,oi) => {
                                const selected = Exam.ans[i] === oi ? 'selected' : '';
                                return `<button onclick="Exam.sel(${i},${oi})" class="opt-btn option-btn w-full text-left p-3 rounded-lg border text-sm flex gap-2 transition ${selected}">
                                    <span class="font-bold opacity-50 w-6">${String.fromCharCode(65+oi)}.</span>
                                    <span class="flex-1 text-left">${window.MathHelper.isMathContent(o) ? window.MathHelper.renderMath(o, 'math-inline') : o}</span>
                                </button>`;
                            }).join('')}
                        </div>
                    </div>`;
                }
                
                const totalPages = Math.ceil(Exam.d.qs.length / questionsPerPage);
                let paginationHTML = '';
                if (totalPages > 1) {
                    paginationHTML = `
                        <div class="flex justify-center gap-2 mt-6">
                            <button onclick="Exam.prevPage()" class="px-4 py-2 bg-slate-100 dark:bg-dark-tertiary rounded-lg ${Exam.currentPage === 0 ? 'opacity-50 cursor-not-allowed' : ''}" ${Exam.currentPage === 0 ? 'disabled' : ''}>
                                <i class="fas fa-chevron-left"></i> Previous
                            </button>
                            <span class="px-4 py-2 text-sm">Page ${Exam.currentPage + 1} of ${totalPages}</span>
                            <button onclick="Exam.nextPage()" class="px-4 py-2 bg-slate-100 dark:bg-dark-tertiary rounded-lg ${Exam.currentPage === totalPages - 1 ? 'opacity-50 cursor-not-allowed' : ''}" ${Exam.currentPage === totalPages - 1 ? 'disabled' : ''}>
                                Next <i class="fas fa-chevron-right"></i>
                            </button>
                        </div>
                    `;
                }
                
                const practiceIndicator = Exam.isPractice ? 
                    `<div class="text-xs bg-amber-100 dark:bg-amber-900 text-amber-800 dark:text-amber-200 px-2 py-1 rounded inline-block mb-2 text-center">
                        <i class="fas fa-flask"></i> Practice Mode
                    </div>` : '';
                
                document.getElementById('app-container').innerHTML = `
                <div class="sticky top-0 bg-white dark:bg-dark-secondary border-b dark:border-dark-tertiary px-4 py-3 flex justify-between items-center z-30 shadow-md">
                    <div>
                        ${practiceIndicator}
                        <div class="text-center">
                            <span class="font-bold block text-sm truncate w-32 dark:text-white mx-auto">${Exam.d.title}</span>
                            <span id="tm" class="text-xs font-mono bg-slate-100 dark:bg-dark-tertiary px-1 rounded text-slate-600 dark:text-slate-300">00:00</span>
                        </div>
                    </div>
                    <div class="flex gap-2">
                        <button onclick="Exam.sub()" class="bg-gradient-to-r from-indigo-500 to-indigo-600 text-white px-5 py-2 rounded-lg text-sm font-bold shadow hover:bg-indigo-700 transition">SUBMIT</button>
                    </div>
                </div>
                <div class="p-4 pb-20 bg-slate-50 dark:bg-black min-h-screen select-none exam-question-container">
                    ${questionHTML}
                    ${paginationHTML}
                </div>`;
                
                await MathJax.typesetPromise();
                
                Exam.updateReviewPanel();
                
                return Promise.resolve();
            },
            
            sel: (qi, oi) => {
                if (Exam.ans[qi] === oi) return;
                
                Exam.ans[qi] = oi;
                
                const questionDiv = document.getElementById(`q-${qi}`);
                if (questionDiv) {
                    const buttons = questionDiv.querySelectorAll('.option-btn');
                    buttons.forEach((btn, index) => {
                        if (index === oi) {
                            btn.classList.add('selected');
                            btn.innerHTML = `
                                <span class="font-bold opacity-50 w-6">${String.fromCharCode(65+index)}.</span>
                                <span class="flex-1 text-left">${Exam.d.qs[qi].options[index]}</span>
                                <i class="fas fa-check text-indigo-600"></i>
                            `;
                        } else {
                            btn.classList.remove('selected');
                            btn.innerHTML = `
                                <span class="font-bold opacity-50 w-6">${String.fromCharCode(65+index)}.</span>
                                <span class="flex-1 text-left">${Exam.d.qs[qi].options[index]}</span>
                            `;
                        }
                    });
                }
                
                Exam.updateReviewPanel();
            },
            
            prevPage: () => {
                if (Exam.currentPage > 0) {
                    Exam.currentPage--;
                    Exam.render();
                    window.scrollTo({ top: 0, behavior: 'smooth' });
                }
            },
            
            nextPage: () => {
                const totalPages = Math.ceil(Exam.d.qs.length / 25);
                if (Exam.currentPage < totalPages - 1) {
                    Exam.currentPage++;
                    Exam.render();
                    window.scrollTo({ top: 0, behavior: 'smooth' });
                }
            },
            
            runT: (sec) => {
                Exam.t = setInterval(()=>{ 
                    sec--; 
                    document.getElementById('tm').innerText=`${Math.floor(sec/60)}:${(sec%60).toString().padStart(2,'0')}`; 
                    if(sec<=0) Exam.sub(true); 
                },1000);
            },
            
            sub: async (auto=false) => {
                if(!auto && !confirm('Submit Exam?')) return;
                clearInterval(Exam.t);
                const neg = Exam.d.negativeMark ? parseFloat(Exam.d.negativeMark) : 0;
                let score = 0;
                Exam.d.qs.forEach((q,i) => { if(Exam.ans[i] == q.correct) score++; else if(Exam.ans[i] !== null) score -= neg; });
                score = Math.max(0, score);
                const showInstant = (Exam.d.type === 'mock' || Exam.d.resultPublished);
                
                await addDoc(collection(db, "attempts"), {
                    userId: auth.currentUser.uid, 
                    userName: AppState.userProfile?.name || auth.currentUser.displayName,
                    examId: Exam.d.id, 
                    examTitle: Exam.d.title, 
                    score, 
                    answers: Exam.ans, 
                    startedAt: Exam.startedAt,
                    submittedAt: new Date(),
                    isPractice: Exam.isPractice || Exam.d.type === 'mock'
                });
                
                if(showInstant) {
                    Swal.fire('Result', `Score: ${score}`, 'success').then(() => { 
                        document.getElementById('review-panel-btn').classList.add('hidden');
                        document.getElementById('review-panel').classList.remove('show');
                        
                        // মক পরীক্ষা শেষে মক রেজাল্ট লিস্ট দেখাবে
                        if (Exam.d.type === 'mock') {
                            window.resultTypeFilter = 'mock';
                            Router.student('results');
                        } else {
                            Router.student('results');
                        }
                    });
                } else {
                    Swal.fire('Submitted', 'Wait for result.', 'success').then(() => {
                        document.getElementById('review-panel-btn').classList.add('hidden');
                        document.getElementById('review-panel').classList.remove('show');
                        Router.student('dashboard');
                    });
                }
                document.getElementById('student-nav').classList.remove('hidden');
            }
        };

        // Setup review panel button
        document.getElementById('review-panel-btn').addEventListener('click', function() {
            document.getElementById('review-panel').classList.toggle('show');
        });

        // Initialize
        onAuthStateChanged(auth, u => {
            if(u) { 
                getDoc(doc(db, "students", u.uid)).then(s=>{ 
                    if(s.exists()) {
                        const userData = s.data();
                        if(userData.blocked) {
                            signOut(auth);
                            Swal.fire('Blocked', 'Your account has been blocked. Please contact admin.', 'error');
                        } else {
                            AppState.user = u;
                            if (userData.disabled) {
                                AppState.userDisabled = true;
                            }
                            
                            AppState.profileCompleted = userData.profileCompleted || false;
                            AppState.userProfile = userData;
                            
                            // Load teacher codes
                            let teacherCodes = userData.teacherCodes || [];
                            if (teacherCodes.length > 0 && typeof teacherCodes[0] === 'string') {
                                // Convert old format to new format
                                teacherCodes = teacherCodes.map(code => ({ code, active: false }));
                                if (teacherCodes.length > 0) teacherCodes[0].active = true;
                            }
                            
                            AppState.teacherCodes = teacherCodes;
                            AppState.activeTeacherCode = teacherCodes.find(tc => tc.active)?.code || null;
                            AppState.groupCode = userData.groupCode || null;
                            AppState.hasGroupCode = !!userData.groupCode;
                            AppState.joinedGroups = userData.joinedGroups || [];
                            
                            // Set active group from localStorage or first joined group
                            const storedGroupId = localStorage.getItem('activeGroupId');
                            if (storedGroupId && AppState.joinedGroups.find(g => g.groupId === storedGroupId)) {
                                AppState.activeGroupId = storedGroupId;
                            } else if (AppState.joinedGroups.length > 0) {
                                AppState.activeGroupId = AppState.joinedGroups[0].groupId;
                                localStorage.setItem('activeGroupId', AppState.activeGroupId);
                            }
                            
                            if (!AppState.profileCompleted) {
                                Router.showProfileForm();
                            } else {
                                Router.initStudent();
                            }
                        }
                    } else {
                        // Create student document if doesn't exist
                        setDoc(doc(db, "students", u.uid), {
                            uid: u.uid,
                            email: u.email,
                            name: "",
                            phone: "",
                            fatherPhone: "",
                            motherPhone: "",
                            schoolName: "",
                            collegeName: "",
                            teacherCodes: [],
                            groupCode: "",
                            joinedGroups: [],
                            profileCompleted: false,
                            blocked: false,
                            disabled: false,
                            joined: new Date()
                        }).then(() => {
                            AppState.profileCompleted = false;
                            Router.showProfileForm();
                        });
                    }
                }); 
            } else { 
                document.getElementById('auth-screen').classList.remove('hidden'); 
            }
        });
    </script>
